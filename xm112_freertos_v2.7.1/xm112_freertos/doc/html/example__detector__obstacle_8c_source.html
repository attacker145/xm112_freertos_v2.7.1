19 /** \example example_detector_obstacle.c
20  * @brief This is an example on how the obstacle detector can be used
21  * @n
22  * The example executes as follows:
23  * - Activate Radar System Software (RSS)
24  * - Create an obstacle detector configuration
25  * - Create an obstacle detector using the previously created configuration
26  * - Activate the obstacle detector
27  * - Optionally do background estimation
28  * - Get the result and print it 5 times
29  * - Deactivate and destroy the obstacle detector
30  * - Destroy the obstacle detector configuration
31  * - Deactivate Radar System Software (RSS)
32  */

61 bool acc_example_detector_obstacle(void)
62 {
63  printf("Acconeer software version %s\n", acc_version_get());
64 
65  hal = acc_driver_hal_get_implementation();

74

77  printf("acc_detector_obstacle_configuration_create failed\n");

97 bool execute_obstacle_detection(acc_detector_obstacle_configuration_t configuration)
98 {
99  acc_detector_obstacle_handle_t handle = acc_detector_obstacle_create(configuration);
100 
101  if (handle == NULL)
102  {
103  printf("acc_detector_obstacle_create() failed\n");
104  return false;
105  }
106 
107  if (!acc_detector_obstacle_activate(handle))
108  {
109  printf("acc_detector_obstacle_activate() failed\n");
110  acc_detector_obstacle_destroy(&handle);
111  return false;
112  }
113 
114  // Handle background estimation differently depending on if it has been done before or ont
115  if (background_estimation_data == NULL)
116  {
117  acc_detector_obstacle_result_info_t result_info;
118  bool completed;
119 
120  do
121  {
122  if (!acc_detector_obstacle_estimate_background(handle, &completed, &result_info))
123  {
124  printf("acc_detector_obstacle_estimate_background() failed\n");
125  acc_detector_obstacle_deactivate(handle);
126  acc_detector_obstacle_destroy(&handle);
127  return false;
128  }
129  } while (!completed);
130 
131  size_t background_estimation_size = acc_detector_obstacle_background_estimation_get_size(handle);
132 
133  background_estimation_data = hal->os.mem_alloc(background_estimation_size);
134  if (background_estimation_data != NULL)
135  {
136  if (!acc_detector_obstacle_background_estimation_get(handle, background_estimation_data))
137  {
138  printf("acc_detector_obstacle_background_estimation_get() failed\n");
139  acc_detector_obstacle_deactivate(handle);
140  acc_detector_obstacle_destroy(&handle);
141  hal->os.mem_free(background_estimation_data);
142  background_estimation_data = NULL;
143  return false;
144  }
145  }
146  }
147  else
148  {
149  if (!acc_detector_obstacle_background_estimation_set(handle, background_estimation_data))
150  {
151  printf("acc_detector_obstacle_background_estimation_get() failed\n");
152  acc_detector_obstacle_deactivate(handle);
153  acc_detector_obstacle_destroy(&handle);
154  hal->os.mem_free(background_estimation_data);
155  background_estimation_data = NULL;
156  return false;
157  }
158  }
159 
160  acc_obstacle_t obstacles[16];
161  acc_detector_obstacle_t obstacle_data;
162  obstacle_data.obstacles = obstacles;
163 
164  bool success = true;
165  const int iterations = 20;
166 
167  for (uint16_t i = 0; i < iterations; i++)
168  {
169  acc_detector_obstacle_result_info_t result_info;
170 
171  do
172  {
173  success = acc_detector_obstacle_get_next(handle, &obstacle_data, &result_info);
174  } while (success && !result_info.data_available);
175 
176  if (!success)
177  {
178  printf("acc_detector_obstacle_get_next() failed\n");
179  break;
180  }
181 
182  uint8_t nbr_of_obstacles = obstacle_data.nbr_of_obstacles;
183  printf("Detected %u obstacles!\n", (unsigned int)nbr_of_obstacles);
184 
185  if (nbr_of_obstacles > 0)
186  {
187  for (uint8_t j = 0; j < nbr_of_obstacles; j++)
188  {
189  printf("Obstacle at a distance %d mm with amplitude %d\n",
190  (int)(obstacle_data.obstacles[j].distance * 1000.0f),
191  (int)(obstacle_data.obstacles[j].amplitude * 1000.0f));
192  }
193  }
194 
195  printf("\n");
196  }
197 
198  bool deactivated = acc_detector_obstacle_deactivate(handle);
199 
200  acc_detector_obstacle_destroy(&handle);
201 
202  hal->os.mem_free(background_estimation_data);
203  background_estimation_data = NULL;
204 
205  return deactivated && success;
206 }

36 /**
37  * @brief Internal function to handle graceful termination of thread when thread function returns
38  *
39  * This is a wrapper function for the main thread function to gracefully
40  * terminate a thread if the thread function returns.
41  *
42  * The data pointed to by thread_info must have been allocated with acc_integration_mem_alloc.
43  *
44  * @param thread_handle A pointer to a acc_app_integration_thread_handle_t struct
45  */
46 static void acc_app_integration_thread_work(void *thread_handle)
47 {
48  acc_app_integration_thread_handle_t thread = (acc_app_integration_thread_handle_t)thread_handle;
49 
50  thread->func(thread->param);
51 
52  xSemaphoreGive(thread->stopped);
53  vTaskDelete(NULL);
54 }

57 acc_app_integration_thread_handle_t acc_app_integration_thread_create(void (*func)(void *param), void *param, const char *name)
58 {
59  assert(func != NULL);
60  BaseType_t result;
61  acc_app_integration_thread_handle_t thread = NULL;
62 
63  thread = pvPortMalloc(sizeof(*thread));
64 
65  if (thread == NULL)
66  {
67  return NULL;
68  }
69 
70  thread->func = func;
71  thread->param = param;
72 
73  thread->stopped = xSemaphoreCreateBinary();
74  if (thread->stopped == NULL)
75  {
76  vPortFree(thread);
77  return NULL;
78  }
79 
80  result = xTaskCreate(acc_app_integration_thread_work, name, ACC_APP_STACK_SIZE / sizeof(int), thread, tskIDLE_PRIORITY + 1,
81  (TaskHandle_t *)&thread->handle);
82  if (result != pdPASS)
83  {
84  vSemaphoreDelete(thread->stopped);
85  vPortFree(thread);
86  return NULL;
87  }
88 
89  return thread;
90 }

121 /**
122  * @brief Convert milliseconds to ticks
123  *
124  * Delaying one tick means "wait until next tick" in FreeRTOS, this means that
125  * the elapsed time for 1 tick can be between 0 and (1 / configTICK_RATE_HZ) seconds
126  *
127  * @param[in] ms Number of milliseconds
128  *
129  * @return The amount of ticks to wait that corresponds to at least ms time
130  */
131 static TickType_t ms_to_ticks(uint32_t ms)
132 {
133  TickType_t ticks = 0;
134 
135  if (ms != 0)
136  {
137  ticks = (uint64_t)ms * (configTICK_RATE_HZ) / 1000 + 1;
138  if ((((uint64_t)ms * (configTICK_RATE_HZ)) % 1000) != 0)
139  {
140  ticks++;
141  }
142  }
143 
144  return ticks;
145 }

174 static bool is_interrupt_context(void)
175 {
176  uint32_t ipsr;
177 
178  __asm("mrs %0, ipsr" : "=r" (ipsr));
179  return ipsr != 0;
180 }

183 void acc_app_integration_semaphore_signal(acc_app_integration_semaphore_t sem)
184 {
185  assert(sem != NULL);
186  if (is_interrupt_context())
187  {
188  BaseType_t xHigherPriorityTaskWoken;
189  xSemaphoreGiveFromISR((SemaphoreHandle_t)sem, &xHigherPriorityTaskWoken);
190  portYIELD_FROM_ISR(xHigherPriorityTaskWoken);
191  }
192  else
193  {
194  xSemaphoreGive((SemaphoreHandle_t)sem);
195  }
196 }

25 /**
26  * @brief Power save mode
27  *
28  * Each power save mode corresponds to how much of the sensor hardware is shutdown
29  * between data frame aquisition. Mode 'OFF' means that the whole sensor is shutdown.
30  * Mode 'ACTIVE' means that the sensor is in its active state all the time.
31  * Mode 'HIBERNATE' means that the sensor is still powered but the internal oscillator is
32  * turned off and the application needs to clock the sensor by toggling a GPIO a pre-defined
33  * number of times to enter and exit this mode. Mode 'HIBERNATE' is currently only supported
34  * by the Sparse service
35  *
36  * For each power save mode there will be a limit in the achievable update rate. Mode 'OFF'
37  * can achieve the lowest power consumption but also has the lowest update rate limit
38  */
39 typedef enum
40 {
41  ACC_POWER_SAVE_MODE_OFF,
42  ACC_POWER_SAVE_MODE_SLEEP,
43  ACC_POWER_SAVE_MODE_READY,
44  ACC_POWER_SAVE_MODE_ACTIVE,
45  ACC_POWER_SAVE_MODE_HIBERNATE,
46 } acc_power_save_mode_enum_t;
47 typedef uint32_t acc_power_save_mode_t;

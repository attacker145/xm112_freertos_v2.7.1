30 typedef struct
31 {
32  bool missed_data; // Indication of missed data from the sensor
33  bool sensor_communication_error; // Indication of a sensor communication error, service probably needs to be restarted
34  bool data_saturated; // Indication of sensor data being saturated, can cause result instability
35  bool data_quality_warning; // Indication of bad data quality that may be addressed by restarting the service to recalibrate the sensor
36  bool data_available; // Indicates if obstacle data is available or if more updates are needed
37  bool proximity_detected; // Indicates that object is detected close to the sensor, requires proximity_detection to be enabled, experimental

58 /**
59  * @brief Activate an obstacle detector.
60  *
61  * Activate an obstacle detector associated by handle. Use acc_detector_obstacle_get_next
62  * to retrieve obstacles.
63  *
64  * Activating an already activated detector has no effect and true will be returned.
65  *
66  * If handle is a nullpointer the activation will fail and return false.
67  *
68  * @param[in] handle A handle for a detector to be activated
69  * @return True if activation succeeded otherwise false.
70  */
71 bool acc_detector_obstacle_activate(acc_detector_obstacle_handle_t handle);
72 
73 
74 /**
75  * @brief Deactivate an obstacle detector.
76  *
77  * Deactivate an obstacle detector associated by the handle.
78  *
79  * Deactivating an already deactivated detector has no effect and true will be returned.
80  * If handle is a nullpointer the activation will fail and return false.
81  *
82  * If the deactivation fails other than if the handle is a nullpointer, the system is in
83  * an undefined state and further activation of detectors and services should be avoided
84  * in the same session.
85  *
86  * @param[in] handle A handle for a detector to be deactivated
87  * @return True if deactivation succeed otherwise false
88  *
89  */
90 bool acc_detector_obstacle_deactivate(acc_detector_obstacle_handle_t handle);

147 /**
148  * @brief Get next update from the sensor and process it
149  *
150  * May only be called after an obstacle detector has been activated to retrieve the next result, blocks
151  * the application until sensor data is ready. Not every call to this function will generate obstacle
152  * information, the result_info->data_available indicates this.
153  * It is important to call this function as often as required in order to not miss any data frames as the
154  * algorithm requires a continuous supply of data updates in order to work properly.
155  *
156  * @param[in] handle The obstacle detector handle for the obstacle detector to get the next result for
157  * @param[out] obstacle_data Information about detected obstacles that is going to be sent to the user.
158  * note that this is only valid if result_info->data_available is set to true.
159  * This parameter can be NULL if no result is wanted.
160  * @param[out] result_info The result info for the last update
161  *
162  * @return True if successful, otherwise false
163  */
164 bool acc_detector_obstacle_get_next(acc_detector_obstacle_handle_t handle,
165  acc_detector_obstacle_t *obstacle_data,
166  acc_detector_obstacle_result_info_t *result_info);

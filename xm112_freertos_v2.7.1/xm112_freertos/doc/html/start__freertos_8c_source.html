138  *
139  * @param[in] file File to write to, ignored
140  * @param[in] ptr Buffer with data to write
141  * @param[in] len Number of bytes to write
142  * @return number of bytes written

145 {
146  (void)file;

151  acc_device_uart_write_buffer(acc_debug_uart_port, ptr, len);

153  }
154 
155  return len;
156 }
